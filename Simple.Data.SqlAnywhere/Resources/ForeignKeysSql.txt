SELECT FK.ROLE CONSTRAINT_NAME, FKTUP.USER_NAME TABLE_SCHEMA , FKT.TABLE_NAME, FKC.COLUMN_NAME, 
       PKTUP.USER_NAME UNIQUE_TABLE_SCHEMA, PKT.TABLE_NAME UNIQUE_TABLE_NAME, PKC.COLUMN_NAME UNIQUE_COLUMN_NAME,
	   PKC.COLUMN_ID ORDINAL_POSITION
FROM SYSFOREIGNKEY FK
JOIN SYSFKCOL FKCOL ON FKCOL.FOREIGN_TABLE_ID = FK.FOREIGN_TABLE_ID AND FKCOL.FOREIGN_KEY_ID=FK.FOREIGN_KEY_ID
JOIN SYSCOLUMN FKC ON FKC.TABLE_ID=FK.FOREIGN_TABLE_ID AND FKC.COLUMN_ID=FKCOL.FOREIGN_COLUMN_ID
JOIN SYSTABLE FKT ON FKT.TABLE_ID=FKC.TABLE_ID
JOIN SYSUSERPERM FKTUP ON FKTUP.USER_ID=FKT.CREATOR
JOIN SYSCOLUMN PKC ON PKC.TABLE_ID=FK.PRIMARY_TABLE_ID AND PKC.COLUMN_ID=FKCOL.PRIMARY_COLUMN_ID
JOIN SYSTABLE PKT ON PKT.TABLE_ID=PKC.TABLE_ID
JOIN SYSUSERPERM PKTUP ON PKTUP.USER_ID=PKT.CREATOR
